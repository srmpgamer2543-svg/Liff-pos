async function openPopup(item) { let basePrice = item.variants[0].default_price; let selectedSweet = null; let selectedType = null; let selectedToppings = []; // ดึง modifier จาก API const modRes = await fetch(BASE_URL + "?path=modifiers"); const ALL_MODIFIERS = await modRes.json(); const itemModifierIds = item.modifier_list_ids || item.modifier_ids || []; const relatedModifiers = ALL_MODIFIERS.filter(m => itemModifierIds.includes(m.modifier_list_id || m.id) ); // แยกตาม group const groups = {}; relatedModifiers.forEach(mod => { const groupName = mod.modifier_list_name; if (!groups[groupName]) groups[groupName] = []; groups[groupName].push(mod); }); const modal = document.createElement("div"); modal.style.position = "fixed"; modal.style.bottom = "0"; modal.style.left = "0"; modal.style.right = "0"; modal.style.background = "white"; modal.style.padding = "20px"; modal.style.borderRadius = "20px 20px 0 0"; modal.style.maxHeight = "80vh"; modal.style.overflowY = "auto"; modal.innerHTML = `

${item.item_name}

`; const totalDiv = document.createElement("div"); function updateTotal() { let total = basePrice; if (selectedSweet) total += parseFloat(selectedSweet.price || 0); if (selectedType) total += parseFloat(selectedType.price || 0); selectedToppings.forEach(t => total += parseFloat(t.price || 0)); totalDiv.innerHTML = "รวม: " + total + " บาท"; addBtn.innerHTML = "เพิ่มลงตะกร้า ("+total+" บาท)"; } Object.keys(groups).forEach(groupName => { const div = document.createElement("div"); div.innerHTML = "

"+groupName+"

"; groups[groupName].forEach(mod => { const input = document.createElement("input"); // ถ้าเป็น group ที่เลือกได้ 1 อัน if (groupName === "ระดับความหวาน" || groupName === "เย็น/ปั่น") { input.type = "radio"; input.name = groupName; } else { input.type = "checkbox"; } input.onchange = () => { if (groupName === "ระดับความหวาน") { selectedSweet = mod; } else if (groupName === "เย็น/ปั่น") { selectedType = mod; } else { if (input.checked) selectedToppings.push(mod); else selectedToppings = selectedToppings.filter(x => x.id !== mod.id); } updateTotal(); }; div.appendChild(input); div.append(" " + mod.modifier_name + " +" + mod.price); div.appendChild(document.createElement("br")); }); modal.appendChild(div); }); const addBtn = document.createElement("button"); addBtn.style.background = "#16a34a"; addBtn.style.color = "white"; addBtn.style.width = "100%"; addBtn.style.padding = "12px"; addBtn.style.borderRadius = "10px"; addBtn.onclick = () => { if (!selectedSweet) return alert("กรุณาเลือกระดับความหวาน"); if (!selectedType) return alert("กรุณาเลือก เย็น หรือ ปั่น"); let final = basePrice; if (selectedSweet) final += parseFloat(selectedSweet.price || 0); if (selectedType) final += parseFloat(selectedType.price || 0); selectedToppings.forEach(t => final += parseFloat(t.price || 0)); addToCart(final); document.body.removeChild(modal); }; modal.appendChild(totalDiv); modal.appendChild(addBtn); document.body.appendChild(modal); updateTotal(); }
